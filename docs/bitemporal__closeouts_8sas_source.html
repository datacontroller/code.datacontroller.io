1/**
2 @file
3 @brief Closes out records
4 @details Closes out records from a temporal table by reference to a single
5 temporal range + business key. Only live records are closed out, so the
6 entire key should be provided in the input table EXCEPT the TECH_FROM.
7 All records matching the key (as per the input table) are closed out
8 on TECH_TO.
9
10 Returns an updated base table and &mpelib..mpe_dataloads table
11
12 potential improvements - write the update statements as a text file and retain
13 for future reference!
14
15 @param [in] now= (DEFINE) Allows consistent tracking of tech dates. Should be
16 a date literal, not a numeric constant, for DB compatibility.
17 @param [in] load_type= Set to UPDATE if non-temporal, else assumed
18 to be TXTEMPORAL. Note that BITEMPORAL is treated the same as TXTEMPORAL
19 given that BUS_FROM should be supplied in the PK.
20 @param [in] tech_from= (tx_from_dttm) Technical FROM datetime variable.
21 Required on BASE table only.
22
23 <h4> Global Variables </h4>
24 @li `dc_dttmtfmt`
25

28 @li mp_abort.sas
29 @li mf_existvar.sas

46%macro bitemporal_closeouts(
47 tech_from=tx_from_dttm
48 ,tech_to = tx_to_dttm /* Technical TO datetime variable.
49 Req'd on BASE table only. */
50 ,base_lib=WORK /* Libref of the BASE table. */
51 ,base_dsn=BASETABLE /* Name of BASE table. */
52 ,append_lib=WORK /* Libref of the STAGING table. */
53 ,append_dsn=APPENDTABLE /* Name of STAGING table. */
54 ,PK= name sex /* Business key, space separated. */
55 /* Should INCLUDE BUS_FROM field if relevant. */
56 ,NOW=DEFINE
57 ,FILTER= /* supply a filter to limit the update */
58 ,outdest= /* supply an unquoted filepath/filename.ext to get
59 a text file containing the update statements */
60 ,loadtype=
61 ,loadtarget=YES /* if <> YES will return without changing anything */
62);
63%put ENTERING &sysmacroname;
64%local x var start;
65%let start=%sysfunc(datetime());
66%dc_assignlib(WRITE,&base_lib)
67%dc_assignlib(WRITE,&append_lib)
68
69%if &now=DEFINE %then %let now=&dc_dttmtfmt.;
70%put &=now;
71/**
72 * perform basic checks
73 */
74/* do tables exist? */
75%if not %sysfunc(exist(&base_lib..&base_dsn)) %then %do;
76 %mp_abort(msg=&base_lib..&base_dsn does not exist)
77%end;
78%else %if %sysfunc(exist(&append_lib..&append_dsn))=0
79 and %sysfunc(exist(&append_lib..&append_dsn,VIEW))=0 %then %do;
80 %mp_abort(msg=&append_lib..&append_dsn does not exist)
81%end;
82/* do TX columns exist? */
83%if &loadtype ne UPDATE %then %do;
84 %if not %mf_existvar(&base_lib..&base_dsn,&tech_from) %then %do;
85 %mp_abort(msg=&tech_from does not exist on &base_lib..&base_dsn)
86 %end;
87 %else %if not %mf_existvar(&base_lib..&base_dsn,&tech_to) %then %do;
88 %mp_abort(msg=&tech_to does not exist on &base_lib..&base_dsn)
89 %end;
90%end;
91/* do PK columns exist? */
92%do x=1 %to %sysfunc(countw(&PK));
93 %let var=%scan(&pk,&x,%str( ));
94 %if not %mf_existvar(&base_lib..&base_dsn,&var) %then %do;
95 %mp_abort(msg=&var does not exist on &base_lib..&base_dsn)
96 %end;
97 %else %if not %mf_existvar(&append_lib..&append_dsn,&var) %then %do;
98 %mp_abort(msg=&var does not exist on &append_lib..&append_dsn)
99 %end;
100%end;
101/* check uniqueness */
102proc sort data=&append_lib..&append_dsn
103 out=___closeout1 noduprecs dupout=___closeout1a;
104 by &pk;
105run;
106%if %mf_getattrn(___closeout1a,NLOBS)>0 %then
107 %put NOTE: dups on (&PK) in (&append_lib..&append_dsn);
108/* is &NOW value within a tolerance? Should not allow renegade closeouts.. */
109%local gap;
110%let gap=0;
111data _null_;
112 now=&now;
113 gap=intck('HOURS',now,datetime());
114 call symputx('gap',gap,'l');
115run;
116%mf_abort(
117 iftrue=(&gap > 24),
118 msg=NOW variable (&now) is not within a 24hr tolerance
119)
120
121/* have any warnings / errs occurred thus far? If so, abort */
122%mf_abort(
123 iftrue=(&syscc>0),
124 msg=Aborted due to SYSCC=&SYSCC status
125)
126
127/**
128 * Create closeout statements. These are sent as individual SQL statements
129 * to ensure pass-through utilisation. The update_cnt variable monitors
130 * how many records were actually updated on the target table.
131 */
132%local update_cnt;
133%let update_cnt=0;
134filename tmp temp;
135data _null_;
136 set ___closeout1;
137 file tmp;
138 if _n_=1 then put 'proc sql noprint;' ;
139 length string $32767.;
140 %if &loadtype=UPDATE %then %do;
141 put "delete from &base_lib..&base_dsn where 1";
142 %end;
143 %else %do;
144 now=symget('now');
145 put "update &base_lib..&base_dsn set &tech_to= " now @;
146 %if %mf_existvar(&base_lib..&base_dsn,PROCESSED_DTTM) %then %do;
147 put " ,PROCESSED_DTTM=" now @;
148 %end;
149 put " where " now " lt &tech_to ";
150 %end;
151 %do x=1 %to %sysfunc(countw(&PK));
152 %let var=%scan(&pk,&x,%str( ));
153 %if %mf_getvartype(&base_lib..&base_dsn,&var)=C %then %do;
154 /* use single quotes to avoid ampersand resolution in data */
155 string=" & &var='"!!trim(prxchange("s/'/''/",-1,&var))!!"'";
156 %end;
157 %else %do;
158 string=cats(" & &var=",&var);
159 %end;
160 put string;
161 %end;
162 put "&filter ;";
163 put '%let update_cnt=%eval(&update_cnt+&sqlobs);%put update_cnt=&update_cnt;';
164run;
165
166data _null_;
167 infile tmp;
168 input;
169 putlog _infile_;
170run;
171
172%if &loadtarget ne YES %then %return;
173
174/* ensure we have a lock */
175%mp_lockanytable(LOCK,
176 lib=&base_lib,ds=&base_dsn
177 ,ref=bitemporal_closeouts
178 ,ctl_ds=&mpelib..mpe_lockanytable
179)
180
181options source2;
182%inc tmp;
183
184filename tmp clear;
185
186/**
187 * Update audit tracker
188 */
189
190%local newobs; %let newobs=%mf_getattrn(work.___closeout1,NLOBS);
191%local user; %let user=%mf_getuser();
192proc sql;
193insert into &mpelib..mpe_dataloads
194 set libref=%upcase("&base_lib")
195 ,DSN=%upcase("&base_dsn")
196 ,ETLSOURCE="&append_lib..&append_dsn contained &newobs records"
197 ,LOADTYPE="CLOSEOUT"
198 ,DELETED_RECORDS=&update_cnt
199 ,NEW_RECORDS=0
200 ,DURATION=%sysfunc(datetime())-&start
201 ,USER_NM="&user"
202 ,PROCESSED_DTTM=&now;
203quit;
204
205
206%mend bitemporal_closeouts;
