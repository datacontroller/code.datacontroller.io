2 @file
3 @brief Downloads data in a variety of formats
4 @details To enable direct download, this service runs in a dedicated stream
5 as a GET request using URL parameters as inputs.
6
7 The inputs are:
8
9 @li table - the libds of the table to be downloaded
10 @li type - either SAS, CSV, EXCEL, MARKDOWN, WEBCSV or WEBTAB
11 @li filter - the filter RK if used

34%global table type filter ds format is_fmt txfrom txto;
35%mpeinit()

37%let user=%mf_getuser();
38%let is_fmt=0;
39
40%mp_abort(iftrue= (%mf_verifymacvars(type table)=0)
41 ,mac=&_program..sas
42 ,msg=%str(Invalid inputs: type table)
43)
44
45%let libds=%upcase(&table); /* actual source */
46%let table=%upcase(&table); /* used as label for fmt catalogs */
47%let lib=%scan(&table,1,.);
48%let ds=%scan(&table,2,.);
49%dc_assignlib(READ,&lib)
50
51data _null_;
52 set &mpelib..MPE_TABLES;
53 where upcase(libref)="&lib" and upcase(dsn)="&ds";
54
55 /* if a TXTEMPORAL table then filter as such */
56 call symputx('txfrom',var_txfrom);
57 call symputx('txto',var_txto);
58
59 ds=symget('ds');
60 is_fmt=0;
61 if subpad(cats(reverse(ds)),1,3)=:'CF-' then do;
62 ds=scan(ds,1,'-');
63 table=cats("&lib..",ds);
64 putlog "Format Catalog Captured";
65 is_fmt=1;
66 call symputx('libds','work.fmtextract');
67 call symputx('table',table);
68 end;
69 call symputx('is_fmt',is_fmt);
70 putlog (_all_)(=);
71run;
72
73%mp_cntlout(
74 iftrue=(&is_fmt=1)
75 ,libcat=&table
76 ,fmtlist=0
77 ,cntlout=work.fmtextract
78)

103%macro mpestp_getrawdata();
104 %local outfile;
105 %if &type=SAS %then %do;
106 %let outfile=%sysfunc(pathname(work))/&table..sas;
107 %mp_ds2cards(base_ds=staged
108 , tgt_ds=&table
109 , cards_file= "&outfile"
110 , maxobs=100000)
111 %let ext=sas;
112 %let mimetype=text;
113 %end;
114 %else %if &type=CSV or (&type=EXCEL and %mf_existfeature(EXPORTXLS) ne 1)
115 /* cannot proc export excel if PC Files is not licensed */
116 %then %do;
117 %let outfile=%sysfunc(pathname(work))/&table..csv;
118 PROC EXPORT DATA= staged
119 OUTFILE= "&outfile"
120 DBMS=csv REPLACE;
121 RUN;
122 %let ext=csv;
123 %let mimetype=csv;
124 %end;
125 %else %if &type=EXCEL %then %do;
126 %let ext=xlsx;
127 %let outfile=%sysfunc(pathname(work))/&table..&ext;
128 PROC EXPORT DATA= staged
129 OUTFILE= "&outfile"
130 DBMS=xlsx ;
131 RUN;
132 %let mimetype=XLSX;
133 %end;
134 %else %if &type=MARKDOWN %then %do;
135 %let ext=md;
136 %let outfile=%sysfunc(pathname(work))/&table..&ext;
137 filename mdref "&outfile" lrecl=32767;
138
139 %mp_ds2md(staged,outref=mdref,showlog=NO)
140
141 %let mimetype=MARKDOWN;
142 %end;
143 %else %if &type=WEBCSV %then %do;
144 PROC EXPORT DATA= staged
145 OUTFILE= _webout
146 DBMS=csv REPLACE;
147 RUN;
148 /* don't set headers */
149 %return;
150 %end;
151 %else %if &type=WEBTAB %then %do;
152 PROC EXPORT DATA= staged
153 OUTFILE= _webout
154 DBMS=tab REPLACE;
155 RUN;
156 /* don't set headers */
157 %return;
158 %end;
159 %else %do;
160 %mp_abort(msg=type &type not supported,mac=mpestp_getrawdata.sas);
161 %end;
162
163 %mp_abort(iftrue= (&syscc ne 0)
164 ,mac=&_program..sas
165 ,msg=%str(syscc=&syscc)
166 )
167
168 %mp_streamfile(contenttype=&mimetype
169 ,inloc=%str(&outfile)
170 ,outname=&table..&ext
171 )
172
173%mend mpestp_getrawdata;
