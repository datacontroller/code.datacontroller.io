22%let ds=%mf_getvalue(work.iwant,libds);
23%dc_assignlib(READ,%scan(&ds,1,.))
24
25proc contents noprint data=&ds
26 out=droplist1 (keep=name type length label varnum format:);
27run;
28data cols(keep=name type length varnum format label);
29 set droplist1(rename=(format=format2 type=type2));
30 name=upcase(name);
31 if type2=2 then do;
32 length format $49.;
33 if format2='' then format=cats('$',length,'.');
34 else if formatl=0 then format=cats(format2,'.');
35 else format=cats(format2,formatl,'.');
36 type='C';
37 ddtype='CHARACTER';
38 end;
39 else do;
40 if format2='' then format=cats(length,'.');
41 else if formatl=0 then format=cats(format2,'.');
42 else if formatd=0 then format=cats(format2,formatl,'.');
43 else format=cats(format2,formatl,'.',formatd);
44 type='N';
45 if format=:'DATETIME' then ddtype='DATETIME';
46 else if format=:'DATE' then ddtype='DATE';
47 else if format=:'TIME' then ddtype='TIME';
48 else ddtype='NUMERIC';
49 end;
50 if label='' then label=name;
51run;
52
53%mp_abort(iftrue= (&syscc ne 0)
