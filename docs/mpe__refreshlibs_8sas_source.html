18%macro mpe_refreshlibs();
19
20%dc_getlibs(outds=work.mm_getLibs)
21proc sort data=mm_getlibs;
22 by libraryref libraryname;
23run;
24
25data libs0;
26 set mm_getlibs;
27 by libraryref;
28 if "%mf_getplatform()"="SASMETA" then do;
29 /* note - invalid libraries can result in exception errors. If this happens,
30 configure the dc_viewlib_check variable to NO in Data Controller Settings */
31 rc=libname(libraryref,,'meta',cats('library="',libraryname,'";'));
32 drop rc;
33 if rc ne 0 then do;
34 putlog "NOTE: Library " libraryname " does not exist!!";
35 putlog (_all_) (=);
36 delete;
37 end;
38 end;
39 if not first.libraryref then delete;
40run;
41
42proc sql;
43create table libs1 as
44 select distinct libname
45 ,engine
46 ,path
47 ,level
48 ,sysname
49 ,sysvalue
50 from dictionary.libnames
51 order by libname, level,engine,path;
52
53data libs2;
54 set libs1;
55 length tran $1024;
56 if missing(sysname) then sysname='Missing';
57 select(sysname);
58 when('Access Permission') tran='Permissions';
59 when('Owner Name') tran='Owner';
60 when('Schema/Owner') tran='schema';
61 otherwise tran=sysname;
62 end;
63run;
64
65proc transpose data=libs2 out=libs3;
66by libname level engine path;
67var sysvalue;
68id tran;
69run;
70
71data libs4(rename=(libname=libref));
72 length paths $8192 perms owners schemas $500 permissions owner schema $1024;
73 if _n_=1 then call missing (of _all_);
74 set libs3;
75 by libname;
76 if engine='V9' then engine='BASE';
77 if first.libname then do;
78 retain paths perms owners schemas;
79 paths='('!!quote(trim(path));
80 perms=permissions;
81 owners=owner;
82 schemas=schema;
83 end;
84 else do;
85 paths=trim(paths)!!' '!!quote(trim(path));
86 perms=trim(perms)!!','!!trim(permissions);
87 owners=trim(owners)!!','!!trim(owner);
88 schemas=trim(schemas)!!' '!!trim(schema);
89 end;
90 if last.libname then do;
91 paths=trim(paths)!!')';
92 schemas=cats(schemas);
93 output;
94 end;
95 keep libname engine paths perms owners schemas;
96run;
97
98proc sql;
99create table libs5 as
100 select a.libref
101 ,coalescec(b.engine,a.engine) as engine length=32
102 ,b.libraryname as libname
103 ,a.paths
104 ,a.perms
105 ,a.owners
106 ,a.schemas
107 ,b.libraryid as libid
108 from libs4 a
109 left join libs0 b
110 on upcase(a.libref)=upcase(b.libraryref)
111 where libref not in ('SASWORK','WORK','SASUSER','CASUSER','TEMP','STPSAMP'
112 ,'MAPSGFK');
113
114%bitemporal_dataloader(base_lib=&dc_libref
115 ,base_dsn=MPE_DATACATALOG_LIBS
116 ,append_dsn=libs5
117 ,PK=LIBREF
118 ,etlsource=&_program
119 ,loadtype=TXTEMPORAL
120 ,tech_from=TX_FROM
121 ,tech_to=TX_TO
122 ,dclib=&dc_libref
123)
124
125%mend mpe_refreshlibs;
