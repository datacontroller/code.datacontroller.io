17%macro mpe_runhook(hookvar);
18%local pgmloc pgmtype;
19%let pgmtype=0;
20%put &sysmacroname: &=hookvar;
21%if %length(&&&hookvar)>0 %then %do;
22 %put &sysmacroname: Executing &&&hookvar;
23 data _null_;
24 rule_value=symget("&hookvar");
25 if scan(upcase(rule_value),-1,'.')='SAS' then do;
26 call symputx('pgmtype','PGM');
27 call symputx('pgmloc',rule_value);
28 end;
29 else do;
30 apploc="%mf_getapploc()";
31 if substr(rule_value,1,1) ne '/'
32 then rule_value=cats(apploc,'/',rule_value);
33 call symputx('pgmloc',rule_value);
34 call symputx('pgmtype','JOB');
35 end;
36 run;

38 %if &pgmtype=PGM %then %do;

40 %end;
