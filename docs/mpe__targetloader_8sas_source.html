26%macro mpe_targetloader(libds= /* library.dataset to LOAD (target) */
27 ,now= %sysfunc(datetime()) /* static processed timestamp */
28 ,etlsource= /* process from whence the data came */
29 ,STAGING_DS= STAGING_DS /* name of staging (work) dataset which should
30 be appended into the target. */
31 ,LOADTARGET=NO /* set to yes to actually load the target */
32 ,CLOSE_VARS= /* provide close vars to override defaults */
33 ,dclib=NOTPROVIDED
34 ,mdebug=0
35 ,dc_dttmtfmt=E8601DT26.6
36 );
37%local lib ds nobs;
38
39/**
40 * if a format catalog (suffix "-FC") we assume the catalog has already been
41 * created by the calling program with a libds of work.fmtextract
42*/
43%let orig_lib=%upcase(%scan(&libds,1,.));
44%let orig_ds=%upcase(%scan(&libds,2,.));
45%let orig_libds=&libds;
46%if %scan(&libds,2,-)=FC %then %do;
47 %let lib=WORK;
48 %let ds=FMTEXTRACT;
49 %let libds=&lib..&ds;
50%end;
51%else %do;
52 %let lib=&orig_lib;
53 %let ds=&orig_ds;
54%end;
55
56%mp_abort(iftrue= (&dclib=NOTPROVIDED)
57 ,mac=&sysmacroname
58 ,msg=%str(dclib=NOTPROVIDED)
59)
60
61/* get table attributes */
62%let nobs=0;
63data work.sumo_config;
64 set &mpelib..mpe_tables;
65 where &dc_dttmtfmt. lt tx_to
66 and libref="&orig_lib"
67 and dsn="&orig_ds";
68 call symputx('LOADTYPE',loadtype,'l');
69 call symputx('BUSKEY',buskey,'l');
70 call symputx('VAR_TXFROM',var_txfrom,'l');
71 call symputx('VAR_TXTO',var_txto,'l');
72 call symputx('VAR_BUSFROM',var_busfrom,'l');
73 call symputx('VAR_BUSTO',var_busto,'l');
74 call symputx('VAR_PROCESSED',VAR_PROCESSED,'l');
75 call symputx('RK_UNDERLYING',RK_UNDERLYING,'l');
76 %if %length(&CLOSE_VARS)=0 %then %do;
77 call symputx('CLOSE_VARS',CLOSE_VARS,'l');
78 %end;
79 call symputx('nobs',_n_,'l');
80 if missing(AUDIT_LIBDS) then AUDIT_LIBDS="&dclib..MPE_AUDIT";
81 call symputx('AUDIT_LIBDS',AUDIT_LIBDS,'l');
82 put (_all_)(=);
83run;
84
85/* check if table is actually configured to load */
86%if &nobs ne 1 %then %do;
87 proc sql;
88 insert into &mpelib..mpe_loads
89 set USER_NM="%mf_getuser()"
90 ,STATUS='FAILED (BAD DS)'
91 ,CSV_DIR=symget('ETLSOURCE')
92 ,PROCESSED_DTTM=&now;
93%end;
94%mp_abort(iftrue= (&nobs=0)
95 ,mac=&sysmacroname
96 ,msg=%str(Table not registered in &mpelib..mpe_tables)
97)
98%mp_abort(iftrue= (&nobs>1)
99 ,mac=&sysmacroname
100 ,msg=%str(Something is very wrong with the PK in &mpelib..mpe_tables)
101)
102
103
104%if &LOADTYPE=TXTEMPORAL %then %do;
105
106 %bitemporal_dataloader(bus_from=,bus_to= /* explicitly empty*/
107 ,tech_from=&VAR_TXFROM
108 ,tech_to = &VAR_TXTO
109 ,base_lib=&lib
110 ,base_dsn=&ds
111 ,append_lib=WORK
112 ,append_dsn=&STAGING_DS
113 ,high_date='31DEC9999:23:59:59'dt
114 ,PK= &buskey
115 ,ETLSOURCE=&ETLSOURCE
116 ,LOADTYPE=&loadtype
117 ,RK_UNDERLYING=&RK_UNDERLYING
118 ,LOADTARGET=&LOADTARGET
119 ,RK_UPDATE_MAXKEYTABLE=&LOADTARGET
120 ,CLOSE_VARS=&CLOSE_VARS
121 ,processed=&VAR_PROCESSED
122 ,dclib=&dclib
123 ,outds_audit=&AUDIT_LIBDS
124 )
125%end;
126%else %if &loadtype=REPLACE %then %do;
127 %dc_assignlib(WRITE,&lib)
128 %if &LOADTARGET=YES %then %do;
129 %mp_lockanytable(LOCK,lib=&lib,ds=&ds,ref=%str(&etlsource),
130 ctl_ds=&dclib..mpe_lockanytable
131 )
132 data WORK.&STAGING_DS;
133 set WORK.&STAGING_DS;
134 %if %mf_existvar(&libds,&VAR_PROCESSED) %then %do;
135 &VAR_PROCESSED = &now;
136 %end;
137 drop _____DELETE__THIS__RECORD_____;
138 run;
139 proc sql; delete * from &libds;
140 proc append base=&libds data=WORK.&STAGING_DS force nowarn;run;
141
142 %mp_lockanytable(UNLOCK,lib=&lib,ds=&ds,ctl_ds=&dclib..mpe_lockanytable)
143 %end;
144%end;
145%else %if &loadtype=UPDATE %then %do;
146 %bitemporal_dataloader(bus_from=,bus_to=
147 ,tech_from= ,tech_to = /* explicitly empty*/
148 ,base_lib=&lib
149 ,base_dsn=&ds
150 ,append_lib=WORK
151 ,append_dsn=&STAGING_DS
152 ,high_date='31DEC9999:23:59:59'dt
153 ,PK= &buskey
154 ,ETLSOURCE=%superq(etlsource)
155 ,LOADTYPE=UPDATE
156 ,RK_UNDERLYING=&RK_UNDERLYING
157 ,LOADTARGET=&LOADTARGET
158 ,RK_UPDATE_MAXKEYTABLE=&LOADTARGET
159 ,processed=&VAR_PROCESSED
160 ,dclib=&dclib
161 ,outds_audit=&AUDIT_LIBDS
162 )
163%end;
164%else %if &loadtype=FORMAT_CAT %then %do;
165 /**
166 * run mp_formatload
167 * inputs:
168 * - LOADTARGET
169 * - CATALOG
170 * - STAGEDATA
171 * - LOADAUDIT
172 * outputs:
173 * work.outds_add
174 * work.outds_del
175 * work.outds_mod
176 */
177 %mp_loadformat(&orig_libds
178 ,&staging_ds
179 ,loadtarget=&LOADTARGET
180 ,auditlibds=&AUDIT_LIBDS
181 ,locklibds=&dclib..mpe_lockanytable
182 ,delete_col=_____DELETE__THIS__RECORD_____
183 ,outds_add=outds_add
184 ,outds_del=outds_del
185 ,outds_mod=outds_mod
186 ,mdebug=&mdebug
187 )
188%end;
189%else %if &loadtype=BITEMPORAL %then %do;
190 %bitemporal_dataloader(bus_from=&VAR_BUSFROM,bus_to=&VAR_BUSTO
191 ,tech_from=&VAR_TXFROM
192 ,tech_to = &VAR_TXTO
193 ,base_lib=&lib
194 ,base_dsn=&ds
195 ,append_lib=WORK
196 ,append_dsn=&STAGING_DS
197 ,high_date='31DEC9999:23:59:59'dt
198 ,PK= &buskey
199 ,ETLSOURCE=%superq(etlsource)
200 ,LOADTYPE=BITEMPORAL
201 ,RK_UNDERLYING=&RK_UNDERLYING
202 ,LOADTARGET=&LOADTARGET
203 ,RK_UPDATE_MAXKEYTABLE=&LOADTARGET
204 ,CLOSE_VARS=&CLOSE_VARS
205 ,processed=&VAR_PROCESSED
206 ,dclib=&dclib
207 ,outds_audit=&AUDIT_LIBDS
208 )
209%end;
210%else %do;
211 %put WARNING: LOADTYPE &LOADTYPE not supported;
212 %let syscc=4;
213 %mp_abort(msg=LOADTYPE &LOADTYPE not supported,mac=mpe_targetloader.sas)
214%end;
215
216%mp_abort(iftrue= (&syscc ne 0)
217 ,mac=&_program
218 ,msg=%str(syscc=&syscc exiting MPE_TARGETLOADER macro)
219)
220
221%mend mpe_targetloader;
