28proc sql noprint;
29create table out1 (rename=(SUBMITTED_ON_DTTM1=SUBMITTED_ON_DTTM)) as
30 select table_id
31 ,submit_status_cd as REVIEW_STATUS_ID
32 ,SUBMITTED_BY_NM
33 ,cats(base_lib,'.',base_ds) as base_table
34 ,put(submitted_on_dttm,yymmddhhmmss.) as SUBMITTED_ON_DTTM1
35 ,submitted_on_dttm as SUBMITTED_ON_DTTM2
36 ,submitted_reason_txt
37 ,num_of_approvals_required

40 ,base_ds as dsn

42 /* filter out any submits for which approval is already made */

47%macro getapprovals();
48 %local admin_check;
49 select count(*) into: admin_check
50 from groups
51 where groupname="&mpeadmins"
52 or groupname in (
53 select sas_group from &mpelib..mpe_security
54 where libref='*ALL*'
55 and &dc_dttmtfmt. lt tx_to
56 and access_level in ('APPROVE')
57 );
58 %if &admin_check >0 %then %do;
59 create table fromSAS as
60 select distinct * from out1
61 order by SUBMITTED_ON_DTTM2 desc;
62 %end;
63 %else %do;
64 create table fromSAS as
65 select distinct a.*
66 from out1 a
67 inner join &mpelib..mpe_security b
68 on a.libref=b.libref
69 and (a.dsn=b.dsn or b.dsn='*ALL*')
70 and &dc_dttmtfmt. lt b.tx_to
71 and b.ACCESS_LEVEL ='APPROVE'
72 and b.SAS_GROUP in (select groupname from work.groups)
73 order by SUBMITTED_ON_DTTM2 desc;
74 %end;
75%mend getapprovals;
