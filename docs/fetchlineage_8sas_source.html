125%macro fcmpconditional();
126%if &sysver=9.3 and &sysscp=WIN %then %do;
127 /* nothing - as the FCMP function causes an exception err in this case */
128%end;
129%else %do;
130 /* prepare quick func to enable word wrapping of transformations */
131 options cmplib=work.funcs;
132 proc fcmp outlib=work.funcs.macrocore;
133 function wordwrap(str $,cols,splitchar $) $;
134 length outstr $32767 curstr $5000;
135 base=0;
136 put str=;
137 do i=1 to countw(str,' ',' ');
138 curstr=scan(str,i,' ');
139 outstr=trim(outstr)!!' '!!curstr;
140 base=base+length(curstr)+1;
141 if base>cols then do;
142 outstr=cats(outstr,splitchar);
143 base=0;
144 end;
145 end;
146 return (outstr);
147 endsub;
148 run;
149%end;
150%mend fcmpconditional;
