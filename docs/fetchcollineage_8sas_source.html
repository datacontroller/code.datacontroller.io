130  /* prepare quick func to enable word wrapping of transformations */

133  function wordwrap(str $,cols,splitchar $) $;
134  length outstr $32767 curstr $5000;
135  base=0;
136  put str=;
137  do i=1 to countw(str,' ',' ');
138  curstr=scan(str,i,' ');
139  outstr=trim(outstr)!!' '!!curstr;
140  base=base+length(curstr)+1;
141  if base>cols then do;
142  outstr=cats(outstr,splitchar);
143  base=0;
144  end;
145  end;
146  return (outstr);
147  endsub;
148  run;
149 %end;
