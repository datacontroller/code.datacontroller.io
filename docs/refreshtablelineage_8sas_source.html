145 /* this bit is slow due to lookups but given the different table types
146  it was the easiest approach */
147 data work.tables2;
148  length tablename liburi puri $64 libref schemaname $8 ;
149  if _n_=1 then call missing(tablename, liburi, puri, libref, schemaname);
150  set work.tables;
151  drop rc liburi;
152  if metadata_getnasn(tableid,"TablePackage",1,liburi)>0 then do;
153  rc= metadata_getattr(liburi, "SchemaName", SchemaName);
154  if ^missing(SchemaName) then do;
155  if metadata_getnasn(liburi,"UsedByPackages",1,puri)>0 then do;
156  rc=metadata_getattr(puri,"Libref",libref);
157  end;
158  end;
159  else rc=metadata_getattr(liburi,"Libref",libref);
160  libref=upcase(libref);
161  end;
162  rc=metadata_getattr(tableid,"Name",tablename);
163  tablename=upcase(tablename);
164 run;
