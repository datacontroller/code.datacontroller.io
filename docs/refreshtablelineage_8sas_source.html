144/* this bit is slow due to lookups but given the different table types

146data work.tables2;
147 length tablename liburi puri $64 libref schemaname $8 ;
148 if _n_=1 then call missing(tablename, liburi, puri, libref, schemaname);
149 set work.tables;
150 drop rc liburi;
151 if metadata_getnasn(tableid,"TablePackage",1,liburi)>0 then do;
152 rc= metadata_getattr(liburi, "SchemaName", SchemaName);
153 if ^missing(SchemaName) then do;
154 if metadata_getnasn(liburi,"UsedByPackages",1,puri)>0 then do;

156 end;
157 end;
158 else rc=metadata_getattr(liburi,"Libref",libref);

165create table work.augmented as

173create table work.mpe_lineage_tabs as
174 select distinct
175 coalesce(a.jobid,b.jobid) as jobid,
176 coalesce(a.jobname,b.jobname) as jobname,
177 coalesce(a.tableid,"N/A") as srctableid,
178 a.tabletype as srctabletype,
179 a.tablename as srctablename,
180 coalesce(a.libref,'nolib') as srclibref,
181 coalesce(b.tableid,"N/A") as tgttableid,
182 b.tabletype as tgttabletype,
183 b.tablename as tgttablename,
184 coalesce(b.libref,'nolib') as tgtlibref
185 from work.augmented(where=(tabledirection='R')) a
186 full join work.augmented(where=(tabledirection='F')) b
187 on a.jobid=b.jobid
