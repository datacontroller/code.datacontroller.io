69%macro mstp_mpeditorstartup();

85 %local admin_check all_cnt;
86 select count(*) into: admin_check
87 from groups
88 where groupname="&mpeadmins";
89 select count(*) into: all_cnt
90 from &mpelib..mpe_security
91 where &dc_dttmtfmt. lt tx_to
92 and ACCESS_LEVEL in ('EDIT')
93 and libref='*ALL*'
94 and SAS_GROUP in (select groupname from groups);
95 %if &admin_check >0 or &all_cnt>0 %then %do;
96 create table sasDatasets as
97 select distinct libref, dsn
98 from &mpelib..mpe_tables
99 where &dc_dttmtfmt. lt tx_to
100 order by 1;
101 %end;
102 %else %do;
103 create table sasDatasets as
104 select distinct a.libref,a.dsn
105 from &mpelib..mpe_tables a
106 left join &mpelib..mpe_security b
107 on a.libref=b.libref
108 and a.dsn=b.dsn
109 where &dc_dttmtfmt. lt a.tx_to
110 and &dc_dttmtfmt. lt b.tx_to
111 and b.ACCESS_LEVEL in ('EDIT')
112 and b.SAS_GROUP in (select groupname from groups)
113 order by 1;
114 %end;
115%mend mstp_mpeditorstartup;
