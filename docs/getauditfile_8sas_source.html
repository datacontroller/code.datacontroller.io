1/**
2 @file getauditfile.sas
3 @brief Downloads a zip file containing audit info.
4 @details The staging location from the &mpelocapprovals location
5 is zipped and returned as a file download. A user can only request the
6 audit pack if they have EDIT or APPROVE rights on the target table.

28%mpeinit()
29options mprint;
30
31/* security checks */
32%let user=%mf_getuser();
33
34proc sql noprint;
35select cats(base_lib,'.',base_ds) into: libds
36 from &mpelib..mpe_submit
37 where table_id="&table";
38
39%mp_abort(
40 iftrue=(%mf_verifymacvars(libds table)=0)
41 ,mac=&_program
42 ,msg=%str(Missing: libds table)
43)
44
45%mpe_accesscheck(&libds,outds=authEDIT,user=&user,access_level=EDIT);
46%mpe_accesscheck(&libds,outds=authAPP,user=&user,access_level=APPROVE);
47
48%mp_abort(
49 iftrue=(
50 %mf_getattrn(work.authEDIT,NLOBS)=0 & %mf_getattrn(work.authAPP,NLOBS)=0
51 )
52 ,mac=mpestp_audit
53 ,msg=%str(&user not authorised to download audit data for &table)
54)
55
56ods package(ProdOutput) open nopf;
57
58options notes source2 mprint;
59%let table=%unquote(&table);
60%mp_dirlist(outds=dirs, path=&mpelocapprovals/&TABLE);
61data _null_;
62 set dirs;
63 retain str1
64 "ods package(ProdOutput) add file='&mpelocapprovals/&TABLE/";
65 retain str2 "' mimetype='text/plain' path='contents/';";
66 call execute(cats(str1,filename,str2));
67run;
68
69%let archive_path=%sysfunc(pathname(work));
70
71ods package(ProdOutput) publish archive properties
72 (archive_name= "&table..zip" archive_path="&archive_path");
73
74ods package(ProdOutput) close;
75
76%mp_abort(iftrue= (&syscc ne 0)
77 ,mac=&_program..sas
78 ,msg=%nrstr(syscc=&syscc)
79)
80
81/* now serve zip file to client */
82%mp_streamfile(contenttype=ZIP
83 ,inloc=%str(&archive_path/&table..zip)
84 ,outname=&table..zip
85)
86
87%mpeterm()
